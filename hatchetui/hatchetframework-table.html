<div id="user-manager" class="hatchet-table-cnt"></div>

JS

var gild =new HatchetFramework.table('#user-manager', columns, options);

// 添加数据
gild.setData(result);

// 添加按钮
$saveA = $("<a  href='javascript:;'>新增</a>").bind("click", function () {
    parent.createTab(null,'saveCarouse-btn',"新增","saveCarouselFlgure.jhtml");
});
if (gild.getTopButton('saveCarouse') == null) {
    gild.setTopButton("saveCarouse", $saveA);
}

// 分页
if(gild.getTopButton('saveCarouse')==null){
    gild.setTopButton("saveCarouse",$saveA);
}

if (pagination == null || pagination == undefined) {
    var options = {callback: paginationCallBack};
    pagination = new HatchetFramework.Pagination("#user-manager_page", options);
} else {
    pagination.setCurrentPageNumber(currentPageNumber);
}

pagination.setPageCount(result.pages);
pagination.setCount(result.count);

//分页控制
function paginationCallBack() {
    if (pagination != null || pagination != undefined) {
        currentPageNumber = pagination.getCurrentPageNumber();
    }
}

基本参数

new HatchetFramework.table(id, columns, options);

id - 元素id

类型：字符串

columns - 列对应参数

类型：数组

属性 值
id 对应tBody要显示数据的key
name 显示的名称
width 列宽 默认宽度80
dragHeader 列拖拽宽度大小

options - 配置参数

类型：json

属性 值
CheckboxEditor 是否显示复选框 默认为false
fixedHead 固定表单头部 默认为false
fixedCol 固定列数 （从左开始固定多少列） 默认为0
page 是否显示分页 默认为true
skin 'blue-theme'和'gray-theme'两种皮肤 默认为'gray-theme'
