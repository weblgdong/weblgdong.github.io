JS

var columns = [{id: "permissionName", name: "名称"}];
var options = {};

var treeTab = new HatchetFramework.Tabletree('permissionTree', options,columns);
// 添加数据
treeTab.setData(data);
// 添加按钮
$addPermissin = $('<a href="javascript:;">添加权限</a>').bind('click', function () {});

if (treeTab.getTopButton('addPermissin') == null) {
   treeTab.setTopButton('addPermissin', $addPermissin);
}

// 设置默认值
function setDefaultNodeChecked(nodes){
    for (var i=0, l=nodes.length; i < l; i++) {
        var node = nodes[i];
        var value = checkedData[node.id];
        if(value != undefined){
            node.permissionKey=value;
        }
        if(value != undefined){
            treeTab.checkNode(node, true, true,true);
        }
        if(node.isParent){
            setDefaultNodeChecked(node.children);
        }
    }
    return nodes;
}

// 刷新已配置的权限列表
function refreshPermissionSitedGrid(){
    checkedData = [];
    $.each(roleId, function(k, rp) {
        checkedData[rp.permissionId]=rp.rolePermissionId;
    });
}

基本参数

new HatchetFramework.Tabletree(id, options,columns);

id - 元素id

类型：字符串

options - 配置参数

属性 值
data 数据设置
check 节点上是否显示 checkbox
callback 用于捕获 checkbox被勾选 或 取消勾选的事件回调函数
type 展现形式 (tree/table) 默认tree

data参数

  1. simpleData
    • asyn 是否异步 默认值：false
    • idKey 节点数据中保存唯一标识的属性名称 （必填）
    • pIdKey 节点数据中保存其父节点唯一标识的属性名称 （必填）
  2. key
    • name 节点数据保存节点名称的属性名称。(必填)
    • children 节点数据中保存子节点数据的属性名称 默认值："children"
